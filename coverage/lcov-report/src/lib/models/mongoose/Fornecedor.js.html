  logo: {
    type: String,
    default: '/static/avatar.png'
  },
  nome: {
    type: String,
    required: [true, 'Nome is required.'],
    unique: true
  },
  descricao: String,
  ...DocumentDefaultProperties
}
 
const FornecedorSchema = new Schema(

    strict: false

)

  return this._id.toHexString()
})
